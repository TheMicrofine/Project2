CREATE TABLE `net`.`user` (

CREATE TABLE `net`.`web_auth` (
  `id` BIGINT NOT NULL AUTO_INCREMENT,
  `email` VARCHAR(255) NOT NULL,
  `salt` CHAR(64) NOT NULL,
  `hashed_password` CHAR(64) NOT NULL,
  `userId` BIGINT NOT NULL,
  PRIMARY KEY (`id`));
